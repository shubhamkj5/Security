import java.math.BigInteger;
import java.security.MessageDigest;
import java.security.NoSuchAlgorithmException;

public class A3MessageDigest2 {
  public static void main(String[] args) {

    String password = "12345";

    MessageDigest algorithm = null;
    try {
      algorithm = MessageDigest.getInstance("MD5");
    } catch (NoSuchAlgorithmException e) {
      e.printStackTrace();
    }
    final byte[] md5Digest = algorithm.digest(password.getBytes());
    System.out.println(md5Digest.length);

    final BigInteger md5Number = new BigInteger(1, md5Digest);
    final String md5 = md5Number.toString(16);
    System.out.println("password " + password);
    System.out.println(" md5 version is " + md5);

  }
}
